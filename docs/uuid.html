Unique ID creation requires a high quality random # generator, but Math.random() does not guarantee "cryptographic quality". So we feature detect for more robust APIs, normalizing each method to return 128-bits (16 bytes) of random data.

 
  var mathRNG, nodeRNG, whatwgRNG;
¶

Math.random()-based RNG. All platforms, very fast, unknown quality

 
  var _rndBytes = new Array(16);
  mathRNG = function() {
    var r, b = _rndBytes, i = 0;

    for (var i = 0, r; i < 16; i++) {
      if ((i & 0x03) == 0) r = Math.random() * 0x100000000;
      b[i] = r >>> ((i & 0x03) << 3) & 0xff;
    }

    return b;
  }
